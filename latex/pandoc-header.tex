% Pandoc header for the master-thesis PDF build.
% Loaded via --include-in-header by latex/build.sh.

% Use DejaVu for broader Unicode coverage (box-drawing, math symbols,
% subscripts) than Latin Modern provides out of the box. fontspec is
% loaded by xelatex automatically when --pdf-engine=xelatex.
\usepackage{fontspec}
\setmainfont{DejaVu Serif}[Scale=0.95]
\setsansfont{DejaVu Sans}[Scale=0.92]
\setmonofont{DejaVu Sans Mono}[Scale=0.85]

% Required packages for the cover and frontmatter inserts
\usepackage{graphicx}
\usepackage{titling}

% Make sure pandoc's TOC, LoF, LoT appear in frontmatter; pandoc will
% include them where the --toc option and template place them. We still
% switch back to mainmatter after them in a small footer file injected
% after those lists.

% Hyperref is loaded by pandoc's default template; we just tweak it via
% the colorlinks metadata in build.sh, so nothing needed here.

% \autoref{} comes from hyperref, which pandoc's default template loads
% AFTER this include file, so nothing extra is required here.

% Gantt charts.
\usepackage{pgfgantt}

% Slightly tighter list spacing.
\usepackage{enumitem}
\setlist{itemsep=0.25em,topsep=0.4em}

% Allow long URLs to break.
\usepackage{xurl}

% TikZ diagrams (used in chapter 4 results figures).
\usepackage{tikz}
\usetikzlibrary{decorations.pathreplacing,arrows.meta,positioning,bending}

% [H] placement specifier — forces figures exactly where they appear in source.
\usepackage{float}

% Better table rules + allow long tables to span pages cleanly.
\usepackage{booktabs}
\usepackage{longtable}
\usepackage{tabularx}   % fixed-width tables with auto-wrapping X columns
\usepackage{array}      % extended column specs (p, m, b, >{...})

% Code listings: pandoc uses fancyvrb / Highlighting environment by default;
% nothing extra required here.

% Make captions a touch smaller.
\usepackage[font=small,labelfont=bf]{caption}

% Avoid the rare overfull-hbox warnings around long inline code.
\sloppy

% Suppress pandoc's auto-generated title page; our custom cover.tex
% (injected via --include-before-body) is the only cover we want.
\renewcommand{\maketitle}{}

% Suppress LaTeX's automatic section numbering entirely — all numbers
% are already written into the heading text in the Markdown sources.
\setcounter{secnumdepth}{-1}

% ---------------------------------------------------------------------------
% Page headers: institutional logos on every body page.
%   Left  : Telecos (ETSETB) logo
%   Right : UPC logo
% Cover (\thispagestyle{empty}) is unaffected.
% ---------------------------------------------------------------------------
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}
\renewcommand{\headrulewidth}{0.4pt}
\setlength{\headheight}{36pt}
\addtolength{\topmargin}{-12pt}
\lhead{\includegraphics[height=1.0cm]{latex/img/logos/logo_telecos.png}}
\rhead{\includegraphics[height=1.0cm]{latex/img/logos/upclogo.png}}
\cfoot{\thepage}

% Apply the same headers to chapter-opening pages (which would otherwise
% revert to \pagestyle{plain} under the report class).
\fancypagestyle{plain}{%
  \fancyhf{}%
  \renewcommand{\headrulewidth}{0.4pt}%
  \lhead{\includegraphics[height=1.0cm]{latex/img/logos/logo_telecos.png}}%
  \rhead{\includegraphics[height=1.0cm]{latex/img/logos/upclogo.png}}%
  \cfoot{\thepage}%
}
